src/lib/email.ts — Resend client singleton
import { Resend } from 'resend';
import { render } from '@react-email/render';

export const resend = new Resend(process.env.RESEND_API_KEY);

const FROM = {
  booking:  'Baggio Travel <booking@baggiotravel.com>',
  support:  'Baggio Travel <support@baggiotravel.com>',
  noreply:  'Baggio Travel <noreply@baggiotravel.com>',
};

// ── Gửi email xác nhận đặt tour ──
export async function sendBookingConfirm(booking: BookingData) {
  const { data, error } = await resend.emails.send({
    from:    FROM.booking,
    to:      [booking.guestEmail],
    subject: `✅ Baggio Travel nhận được yêu cầu — #${booking.ref}`,
    react:   BookingConfirmEmail({ booking }),
    tags:    [{ name: 'type', value: 'booking-confirm' }],
  });
  if (error) throw error;
  return data;
}

// ── Gửi xác nhận thanh toán ──
export async function sendPaymentSuccess(booking: BookingData, payment: PaymentData) {
  return resend.emails.send({
    from:    FROM.booking,
    to:      [booking.guestEmail],
    subject: `💳 Đã nhận cọc ₫${fmt(payment.amount)} — Tour ${booking.tourName}`,
    react:   PaymentSuccessEmail({ booking, payment }),
    tags:    [{ name: 'type', value: 'payment-success' }],
  });
}

src/api/booking.js — Trigger email ngay sau khi tạo booking
// POST /api/booking/create
export default async function handler(req, res) {
  const booking = await createBooking(req.body);

  // Fire email ngay — không await để không delay response
  sendBookingConfirm(booking).catch(err =>
    console.error('Email failed:', err) // không throw — booking vẫn thành công
  );

  res.json({ success: true, bookingRef: booking.ref });
}
src/api/payment/webhook.js — Trigger sau khi MoMo/VNPay xác nhận
// POST /api/payment/momo/webhook
export default async function handler(req, res) {
  const { resultCode, orderId } = req.body;
  if (resultCode !== 0) { res.json({ message: 'OK' }); return; }

  const booking = await getBookingByOrderId(orderId);
  const payment = await updatePaymentSuccess(orderId);

  sendPaymentSuccess(booking, payment).catch(console.error);

  res.json({ message: 'OK' });
}

// vercel.json — thêm cron jobs
{
  "crons": [
    {
      "path": "/api/cron/reminder",
      "schedule": "0 8 * * *"   // chạy 8:00 SA mỗi ngày
    },
    {
      "path": "/api/cron/review-request",
      "schedule": "0 10 * * *"  // chạy 10:00 SA mỗi ngày
    }
  ]
}

// src/api/cron/reminder.ts
export default async function handler(req, res) {
  if (req.headers['authorization'] !== `Bearer ${process.env.CRON_SECRET}`)
    return res.status(401).end();

  const tomorrow = new Date();
  tomorrow.setDate(tomorrow.getDate() + 2); // 48h trước

  const bookings = await db.booking.findMany({
    where: {
      status:     'CONFIRMED',
      depart_date: { gte: startOfDay(tomorrow), lte: endOfDay(tomorrow) },
      reminder_sent: false,
    },
    include: { user: true, tour: true },
  });

  await Promise.allSettled(
    bookings.map(async b => {
      await sendTourReminder(b);
      await db.booking.update({ where: { id: b.id }, data: { reminder_sent: true } });
    })
  );

  res.json({ sent: bookings.length });
}
