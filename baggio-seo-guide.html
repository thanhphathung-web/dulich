src/api/sitemap.js — Auto-generate từ DBNode.js
// GET /sitemap.xml — Vercel serverless function
export default async function handler(req, res) {
  const tours = await db.tour.findMany({
    where:  { status: 'ACTIVE' },
    select: { slug: true, updated_at: true, cover_image: true, name: true },
  });

  const urls = [
    { loc: '/',           priority: '1.0', freq: 'daily' },
    { loc: '/mien-bac',   priority: '0.9', freq: 'weekly' },
    { loc: '/mien-trung', priority: '0.9', freq: 'weekly' },
    { loc: '/mien-nam',   priority: '0.9', freq: 'weekly' },
    ...tours.map(t => ({
      loc:     `/tour/${t.slug}`,
      lastmod: t.updated_at.toISOString().split('T')[0],
      priority: '0.85',
      freq:    'weekly',
      image:   t.cover_image,
      imgTitle: t.name,
    })),
  ];

  const xml = `<?xml version="1.0" encoding="UTF-8"?>
<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9"
        xmlns:image="http://www.google.com/schemas/sitemap-image/1.1">
${urls.map(u => `  <url>
    <loc>https://baggiotravel.com${u.loc}</loc>
    ${u.lastmod ? `<lastmod>${u.lastmod}</lastmod>` : ''}
    <changefreq>${u.freq}</changefreq>
    <priority>${u.priority}</priority>
    ${u.image ? `<image:image><image:loc>${u.image}</image:loc><image:title>${u.imgTitle}</image:title></image:image>` : ''}
  </url>`).join('\n')}
</urlset>`;

  res.setHeader('Content-Type', 'application/xml');
  res.setHeader('Cache-Control', 's-maxage=86400'); // cache 24h
  res.send(xml);
}

Google Search Console
Setup & Monitoring
Search Console cho biết Google đang crawl trang nào, từ khoá nào mang traffic, và lỗi index cần sửa.
5 bước setup Search Console
1
Vào search.google.com/search-console → Add property
Chọn "Domain" (khuyên dùng) → nhập baggiotravel.com
2
Verify ownership qua DNS TXT record
Google cung cấp TXT record → thêm vào Namecheap/Cloudflare → Verify. Mất 1–48h propagate.
3
Submit Sitemap
Sitemaps → Add → nhập https://baggiotravel.com/sitemap.xml → Submit
4
Request Indexing cho trang quan trọng
URL Inspection → nhập URL trang chủ → Request Indexing. Làm lần lượt cho từng trang tour.
5
Kết nối với GA4
Settings → Linked accounts → Google Analytics → Link với property GA4. Dữ liệu tích hợp sau vài ngày.
📊
Theo dõi hàng tuần sau khi index
Performance report → lọc theo từng URL tour → xem Impressions, Clicks, CTR, Average position. CTR < 2% → cần cải thiện title/description. Position 4–10 → cần tăng backlinks.
