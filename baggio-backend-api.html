import express from 'express';
import cors from 'cors';
import helmet from 'helmet';
import { rateLimit } from 'express-rate-limit';
import { errorHandler } from './middleware/errors';
import toursRouter     from './routes/tours';
import bookingsRouter  from './routes/bookings';
import paymentsRouter  from './routes/payments';
import authRouter      from './routes/auth';
import usersRouter     from './routes/users';
import guidesRouter    from './routes/guides';

const app = express();

// ── Security headers
app.use(helmet());
app.use(cors({
  origin: ['https://baggiotravel.com', 'http://localhost:3000'],
  credentials: true,
}));

// ── Body parsing
app.use('/api/payment/webhook', express.raw({ type: '*/*' })); // raw for webhook sig verify
app.use(express.json({ limit: '10mb' }));

// ── Global rate limit
app.use('/api', rateLimit({
  windowMs: 15 * 60 * 1000, // 15 phút
  max: 100,
  standardHeaders: true,
  message: { error: 'Quá nhiều request. Thử lại sau.' },
}));

// ── Routes
app.use('/api/auth',     authRouter);
app.use('/api/tours',    toursRouter);
app.use('/api/bookings', bookingsRouter);
app.use('/api/payment',  paymentsRouter);
app.use('/api/users',    usersRouter);
app.use('/api/guides',   guidesRouter);

// ── Health check
app.get('/api/health', (_, res) => res.json({
  status: 'ok', version: process.env.npm_package_version, ts: new Date()
}));

// ── Error handler (PHẢI đặt cuối cùng)
app.use(errorHandler);

export default app;
src/routes/auth.ts — Register, Login, Refresh token
import { Router } from 'express';
import bcrypt from 'bcrypt';
import jwt from 'jsonwebtoken';
import { z } from 'zod';
import { db } from '../lib/prisma';
import { validate } from '../middleware/validate';

const router = Router();

const JWT_SECRET  = process.env.JWT_SECRET!;
const JWT_REFRESH = process.env.JWT_REFRESH_SECRET!;

const signTokens = (userId: string, role: string) => ({
  accessToken:  jwt.sign({ sub: userId, role }, JWT_SECRET,  { expiresIn: '15m' }),
  refreshToken: jwt.sign({ sub: userId },       JWT_REFRESH, { expiresIn: '30d' }),
});

// POST /api/auth/register
const RegisterSchema = z.object({
  email:    z.string().email(),
  password: z.string().min(8),
  fullName: z.string().min(2),
  phone:    z.string().regex(/^0[0-9]{9}$/).optional(),
});

router.post('/register', validate(RegisterSchema), async (req, res, next) => {
  try {
    const { email, password, fullName, phone } = req.body;
    const exists = await db.user.findUnique({ where: { email } });
    if (exists) return res.status(409).json({ error: 'Email đã tồn tại' });

    const passwordHash = await bcrypt.hash(password, 12);
    const user = await db.user.create({
      data: { email, password_hash: passwordHash, full_name: fullName, phone },
      select: { id: true, email: true, full_name: true, role: true },
    });
    res.status(201).json({ user, ...signTokens(user.id, user.role) });
  } catch (e) { next(e); }
});

// POST /api/auth/login
const LoginSchema = z.object({ email: z.string().email(), password: z.string() });
router.post('/login', validate(LoginSchema), async (req, res, next) => {
  try {
    const { email, password } = req.body;
    const user = await db.user.findUnique({ where: { email } });
    if (!user || !await bcrypt.compare(password, user.password_hash))
      return res.status(401).json({ error: 'Email hoặc mật khẩu không đúng' });

    await db.user.update({ where: { id: user.id }, data: { last_login_at: new Date() } });
    const { password_hash: _, ...safeUser } = user;
    res.json({ user: safeUser, ...signTokens(user.id, user.role) });
  } catch (e) { next(e); }
});

// POST /api/auth/refresh
router.post('/refresh', async (req, res) => {
  try {
    const { refreshToken } = req.body;
    const payload = jwt.verify(refreshToken, JWT_REFRESH) as { sub: string };
    const user = await db.user.findUnique({ where: { id: payload.sub } });
    if (!user) return res.status(401).json({ error: 'Invalid token' });
    res.json(signTokens(user.id, user.role));
  } catch {
    res.status(401).json({ error: 'Token hết hạn' });
  }
});

export default router;
src/middleware/auth.ts — JWT verify middleware
import jwt from 'jsonwebtoken';
import { RequestHandler } from 'express';

export const authenticate: RequestHandler = (req, res, next) => {
  const auth = req.headers.authorization;
  if (!auth?.startsWith('Bearer '))
    return res.status(401).json({ error: 'Chưa đăng nhập' });

  try {
    const payload = jwt.verify(auth.slice(7), process.env.JWT_SECRET!);
    req.user = payload as { sub: string; role: string };
    next();
  } catch {
    res.status(401).json({ error: 'Token hết hạn hoặc không hợp lệ' });
  }
};

// Role guard
export const requireRole = (...roles: string[]): RequestHandler =>
  (req, res, next) => {
    if (!roles.includes(req.user?.role ?? ''))
      return res.status(403).json({ error: 'Không có quyền truy cập' });
    next();
  };

// Validate body với Zod schema
export const validate = (schema: ZodSchema): RequestHandler =>
  (req, res, next) => {
    const result = schema.safeParse(req.body);
    if (!result.success)
      return res.status(400).json({ error: 'Dữ liệu không hợp lệ', details: result.error.flatten() });
    req.body = result.data;
    next();
  };

Query params: destination, region, maxPrice, minDays, maxDays, page, limit, q (full-text search)
src/routes/tours.ts
import { Router } from 'express';
import { db } from '../lib/prisma';
import { authenticate, requireRole } from '../middleware/auth';
import { validate } from '../middleware/validate';
import { z } from 'zod';

const router = Router();

// GET /api/tours — public
router.get('/', async (req, res, next) => {
  try {
    const { destination, region, maxPrice, minDays, maxDays, q, page = 1, limit = 12 } = req.query;
    const skip = (+page - 1) * +limit;

    const where: Prisma.TourWhereInput = {
      status: 'ACTIVE',
      ...(destination && { destination: { contains: destination as string, mode: 'insensitive' } }),
      ...(region      && { region }),
      ...(maxPrice    && { base_price: { lte: +maxPrice } }),
      ...(minDays     && { duration_days: { gte: +minDays } }),
      ...(maxDays     && { duration_days: { lte: +maxDays } }),
      ...(q           && { OR: [
        { name: { contains: q as string, mode: 'insensitive' } },
        { destination: { contains: q as string, mode: 'insensitive' } },
      ]}),
    };

    const [tours, total] = await Promise.all([
      db.tour.findMany({
        where, skip, take: +limit,
        orderBy: { base_price: 'asc' },
        include: {
          schedules: { where: { status: 'OPEN', depart_date: { gte: new Date() } }, take: 5 },
          reviews:   { where: { is_published: true }, select: { rating: true } },
        },
      }),
      db.tour.count({ where }),
    ]);

    // Tính rating trung bình
    const enriched = tours.map(t => ({
      ...t,
      rating_avg: t.reviews.length
        ? t.reviews.reduce((s, r) => s + r.rating, 0) / t.reviews.length
        : null,
      review_count: t.reviews.length,
    }));

    res.json({
      data:  enriched,
      meta:  { total, page: +page, limit: +limit, pages: Math.ceil(total / +limit) },
    });
  } catch (e) { next(e); }
});

// GET /api/tours/:slug — public
router.get('/:slug', async (req, res, next) => {
  try {
    const tour = await db.tour.findUnique({
      where:   { slug: req.params.slug, status: 'ACTIVE' },
      include: {
        schedules: { where: { status: 'OPEN', depart_date: { gte: new Date() } }, orderBy: { depart_date: 'asc' } },
        addons:    { where: { is_active: true } },
        reviews:   { where: { is_published: true }, include: { user: { select: { full_name: true, avatar_url: true } } }, take: 10, orderBy: { created_at: 'desc' } },
      },
    });
    if (!tour) return res.status(404).json({ error: 'Tour không tồn tại' });
    res.json(tour);
  } catch (e) { next(e); }
});

// POST /api/tours — admin only
router.post('/', authenticate, requireRole('ADMIN', 'STAFF'), async (req, res, next) => {
  try {
    const tour = await db.tour.create({ data: req.body });
    res.status(201).json(tour);
  } catch (e) { next(e); }
});

export default router;
src/routes/bookings.ts — Tạo, xem, huỷ booking
const CreateBookingSchema = z.object({
  tourId:     z.string().uuid(),
  scheduleId: z.string().uuid(),
  adultCount: z.number().int().min(1).max(20),
  childCount: z.number().int().min(0).default(0),
  roomType:   z.enum(['STANDARD', 'DELUXE', 'SUITE']).default('STANDARD'),
  addonIds:   z.array(z.string().uuid()).optional(),
  specialRequests: z.string().max(500).optional(),
});

// POST /api/bookings — authenticated
router.post('/', authenticate, validate(CreateBookingSchema), async (req, res, next) => {
  try {
    const userId = req.user!.sub;
    const { tourId, scheduleId, adultCount, childCount, roomType, addonIds } = req.body;

    const [tour, schedule] = await Promise.all([
      db.tour.findUnique({ where: { id: tourId } }),
      db.tourSchedule.findUnique({ where: { id: scheduleId } }),
    ]);

    if (!tour || !schedule)       return res.status(404).json({ error: 'Tour hoặc lịch không tồn tại' });
    if (schedule.available_slots < adultCount + childCount)
      return res.status(409).json({ error: 'Lịch này đã hết chỗ' });

    // Tính giá
    const price = Number(schedule.price_override ?? tour.base_price);
    const childPrice = price * Number(tour.child_price_ratio);
    const roomExtra  = roomType === 'DELUXE' ? 350000 * schedule.duration_nights :
                       roomType === 'SUITE'  ? 800000 * schedule.duration_nights : 0;

    let addonTotal = 0;
    if (addonIds?.length) {
      const addons = await db.tourAddon.findMany({ where: { id: { in: addonIds } } });
      addonTotal = addons.reduce((s, a) => s + Number(a.price), 0);
    }

    const totalPrice   = price * adultCount + childPrice * childCount + roomExtra + addonTotal;
    const depositAmount = Math.round(totalPrice * 0.3);
    const bookingRef   = `BT-${Date.now().toString(36).toUpperCase()}`;

    const booking = await db.$transaction(async tx => {
      const b = await tx.booking.create({
        data: { booking_ref: bookingRef, user_id: userId, tour_id: tourId, schedule_id: scheduleId,
                adult_count: adultCount, child_count: childCount, room_type: roomType,
                total_price: totalPrice, deposit_amount: depositAmount,
                special_requests: req.body.specialRequests },
      });
      await tx.tourSchedule.update({
        where: { id: scheduleId },
        data:  { available_slots: { decrement: adultCount + childCount } },
      });
      return b;
    });

    sendBookingConfirm({ ...booking, guestName: req.user!.name }).catch(console.error);

    res.status(201).json(booking);
  } catch (e) { next(e); }
});

// GET /api/bookings — danh sách của user hiện tại
router.get('/', authenticate, async (req, res, next) => {
  try {
    const bookings = await db.booking.findMany({
      where: { user_id: req.user!.sub },
      include: { tour: { select: { name: true, cover_image: true, destination: true } } },
      orderBy: { created_at: 'desc' },
    });
    res.json(bookings);
  } catch (e) { next(e); }
});

// DELETE /api/bookings/:id — huỷ tour
router.delete('/:id', authenticate, async (req, res, next) => {
  try {
    const booking = await db.booking.findUnique({ where: { id: req.params.id } });
    if (!booking || booking.user_id !== req.user!.sub) return res.status(404).end();
    if (!['PENDING', 'CONFIRMED'].includes(booking.status))
      return res.status(400).json({ error: 'Không thể huỷ đơn đang trong hành trình' });

    await db.booking.update({
      where: { id: booking.id },
      data:  { status: 'CANCELLED', cancel_reason: req.body.reason, cancelled_at: new Date() },
    });
    res.json({ success: true });
  } catch (e) { next(e); }
});
src/middleware/errors.ts — Global error handler
import { ErrorRequestHandler } from 'express';
import { Prisma } from '@prisma/client';
import { ZodError } from 'zod';

export const errorHandler: ErrorRequestHandler = (err, req, res, next) => {
  console.error(`[${req.method}] ${req.path}`, err);

  // Zod validation error
  if (err instanceof ZodError)
    return res.status(400).json({ error: 'Dữ liệu không hợp lệ', details: err.flatten() });

  // Prisma unique constraint
  if (err instanceof Prisma.PrismaClientKnownRequestError) {
    if (err.code === 'P2002')
      return res.status(409).json({ error: 'Dữ liệu đã tồn tại' });
    if (err.code === 'P2025')
      return res.status(404).json({ error: 'Không tìm thấy' });
  }

  // JWT error
  if (err.name === 'JsonWebTokenError' || err.name === 'TokenExpiredError')
    return res.status(401).json({ error: 'Phiên đăng nhập hết hạn' });

  // Unknown — không expose error detail ra ngoài
  res.status(err.statusCode ?? 500).json({
    error: process.env.NODE_ENV === 'production'
      ? 'Lỗi hệ thống. Vui lòng thử lại sau.'
      : err.message,
  });
};

import { rateLimit } from 'express-rate-limit';

// Auth endpoints — chặt hơn để chống brute force
export const authLimiter = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 10,
  skipSuccessfulRequests: true, // chỉ đếm failed requests
  message: { error: 'Quá nhiều lần đăng nhập thất bại. Thử lại sau 15 phút.' },
});

// Payment endpoints — nghiêm ngặt nhất
export const paymentLimiter = rateLimit({
  windowMs: 60 * 1000, // 1 phút
  max: 5,
  message: { error: 'Quá nhiều yêu cầu thanh toán. Vui lòng chờ.' },
});

// Dùng trong routes:
router.post('/auth/login', authLimiter, ...);
router.post('/payment/create', authenticate, paymentLimiter, ...);
