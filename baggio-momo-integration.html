Tích hợp MoMo Payment API v3
Hướng dẫn tích hợp cổng thanh toán MoMo vào Baggio Travel — từ sandbox đến production, bao gồm tạo đơn, xử lý webhook, hoàn tiền và kiểm tra giao dịch.
ℹ️
Loại tích hợp phù hợp nhất cho Baggio Travel
MoMo cung cấp nhiều loại API. Cho website đặt tour, nên dùng Collection Link (payWithMethod) — hỗ trợ MoMo eWallet, ATM, Visa/Master/JCB trên cùng 1 API endpoint. Endpoint: POST /v2/gateway/api/create

Tạo Signature (HMAC-SHA256)
Mọi request đến MoMo phải có signature để xác thực. Signature được tạo bằng HMAC-SHA256 từ chuỗi raw data theo thứ tự cố định.
🔑
Công thức tạo Signature
rawSignature = "accessKey=&AMOUNT&extraData=&ipnUrl=&orderId=&orderInfo=&partnerCode=&redirectUrl=&requestId=&requestType="
signature = HMAC_SHA256(rawSignature, secretKey)
utils/momo-signature.js javascript 
const crypto = require('crypto');

// Tạo signature cho MoMo API request
function createMoMoSignature(params, secretKey) {
  const rawSignature = [
    `accessKey=${params.accessKey}`,
    `amount=${params.amount}`,
    `extraData=${params.extraData}`,
    `ipnUrl=${params.ipnUrl}`,
    `orderId=${params.orderId}`,
    `orderInfo=${params.orderInfo}`,
    `partnerCode=${params.partnerCode}`,
    `redirectUrl=${params.redirectUrl}`,
    `requestId=${params.requestId}`,
    `requestType=${params.requestType}`,
  ].join('&');

  return crypto
    .createHmac('sha256', secretKey)
    .update(rawSignature)
    .digest('hex');
}

// Xác thực signature từ webhook IPN
function verifyWebhookSignature(body, secretKey) {
  const { signature, ...rest } = body;
  const rawData = Object.keys(rest)
    .sort()
    .map(k => `${k}=${rest[k]}`)
    .join('&');
  const expected = crypto
    .createHmac('sha256', secretKey)
    .update(rawData)
    .digest('hex');
  return signature === expected;
}

module.exports = { createMoMoSignature, verifyWebhookSignature };
API: Tạo thanh toán POST
Endpoint chính để tạo đơn thanh toán MoMo và nhận về payUrl để redirect khách hàng.
Request Parameters
Tham sốKiểuMô tả
partnerCode*StringMã partner, nhận từ M4B Portal (VD: MOMO cho sandbox)
requestId*String(50)ID request duy nhất, dùng cho idempotency. Nên dùng UUID v4.
Không được trùng giữa các request
amount*LongSố tiền thanh toán (VND). Min: 1.000đ, Max: 50.000.000đ/giao dịch
orderId*String(64)Mã đơn hàng của merchant. Unique, không đoán được. VD: BT-0891-1717123456
orderInfo*StringMô tả hiển thị cho khách hàng trên app MoMo. VD: "Đặt cọc Tour Hạ Long 5N4Đ - 2 khách"
redirectUrl*StringURL redirect sau khi khách hoàn tất thanh toán. VD: https://baggio.travel/booking/success
ipnUrl*StringWebhook URL nhận IPN notification từ MoMo. Phải là HTTPS và public accessible.
Timeout tối thiểu 30s. MoMo retry 5 lần nếu không nhận được 200 OK
requestType*StringLoại request. Dùng payWithMethod cho Collection Link (hỗ trợ tất cả PTTT)
extraData*StringDữ liệu thêm (base64). Dùng "" nếu không cần. Có thể nhúng orderId tour để callback.
signature*StringHMAC-SHA256 signature. Xem phần "Tạo Signature" ở trên.
itemsArrayDanh sách sản phẩm (optional). Hiển thị chi tiết trên trang thanh toán MoMo.
routes/payment.js node.js · express 
const express = require('express');
const axios   = require('axios');
const { v4: uuidv4 } = require('uuid');
const { createMoMoSignature } = require('../utils/momo-signature');
const router = express.Router();

// Cấu hình — lưu trong .env, không hardcode
const MOMO_CONFIG = {
  partnerCode: process.env.MOMO_PARTNER_CODE, // 'MOMO' cho sandbox
  accessKey:   process.env.MOMO_ACCESS_KEY,
  secretKey:   process.env.MOMO_SECRET_KEY,
  apiUrl:      process.env.MOMO_API_URL,  // test-payment.momo.vn hoặc payment.momo.vn
  redirectUrl: `${process.env.APP_URL}/booking/success`,
  ipnUrl:      `${process.env.APP_URL}/api/payment/momo/webhook`,
};

// POST /api/payment/momo/create
router.post('/create', async (req, res) => {
  try {
    const { bookingId, amount, tourName, guestCount } = req.body;

    // Validate
    if (!bookingId || !amount || amount < 1000) {
      return res.status(400).json({ error: 'Invalid booking data' });
    }

    const requestId = uuidv4();
    const orderId   = `BT-${bookingId}-${Date.now()}`;
    const orderInfo = `Đặt cọc tour ${tourName} - ${guestCount} khách`;
    const extraData = Buffer.from(JSON.stringify({ bookingId })).toString('base64');

    const params = {
      partnerCode: MOMO_CONFIG.partnerCode,
      accessKey:   MOMO_CONFIG.accessKey,
      requestId,
      amount:      String(amount),
      orderId,
      orderInfo,
      redirectUrl: MOMO_CONFIG.redirectUrl,
      ipnUrl:      MOMO_CONFIG.ipnUrl,
      requestType: 'payWithMethod',
      extraData,
    };

    params.signature = createMoMoSignature(params, MOMO_CONFIG.secretKey);

    // Lưu requestId vào DB để idempotency check
    await db.booking.update({
      where: { id: bookingId },
      data: { momoRequestId: requestId, momoOrderId: orderId }
    });

    const { data } = await axios.post(
      `https://${MOMO_CONFIG.apiUrl}/v2/gateway/api/create`,
      params,
      { timeout: 30000 }  // timeout tối thiểu 30s theo docs
    );

    if (data.resultCode !== 0) {
      throw new Error(`MoMo error ${data.resultCode}: ${data.message}`);
    }

    res.json({
      payUrl:    data.payUrl,      // redirect khách hàng đến đây
      qrCodeUrl: data.qrCodeUrl,  // hiển thị QR nếu muốn
      deeplink:  data.deeplink,   // mở thẳng app MoMo trên mobile
      orderId,
    });

  } catch (err) {
    console.error('MoMo payment error:', err.message);
    res.status(500).json({ error: 'Payment creation failed' });
  }
});

module.exports = router;
Webhook IPN (Instant Payment Notification)
MoMo gọi ipnUrl ngay sau khi giao dịch hoàn tất. Đây là cách đáng tin cậy nhất để cập nhật trạng thái đơn hàng — không phụ thuộc vào redirect.
🔒
Bắt buộc xác thực signature webhook
Luôn verify signature của mọi IPN request trước khi xử lý. Kẻ tấn công có thể giả mạo webhook để cập nhật trạng thái thanh toán giả.
routes/payment.js (tiếp theo) node.js · webhook handler 
// POST /api/payment/momo/webhook  ← ipnUrl
router.post('/webhook', async (req, res) => {
  const body = req.body;

  // 1. Xác thực signature TRƯỚC khi làm gì khác
  const isValid = verifyWebhookSignature(body, MOMO_CONFIG.secretKey);
  if (!isValid) {
    console.warn('⚠ Invalid MoMo webhook signature', body);
    return res.status(400).json({ message: 'Invalid signature' });
  }

  const {
    orderId, resultCode, transId,
    amount, message, extraData
  } = body;

  // 2. Parse extraData để lấy bookingId
  const { bookingId } = JSON.parse(
    Buffer.from(extraData, 'base64').toString('utf8')
  );

  // 3. Idempotency — tránh xử lý 2 lần nếu MoMo retry
  const booking = await db.booking.findUnique({ where: { id: bookingId } });
  if (booking.paymentStatus === 'PAID') {
    return res.json({ message: 'Already processed' }); // 200 OK để MoMo không retry
  }

  if (resultCode === 0) {
    // ✅ Thanh toán thành công
    await Promise.all([
      db.booking.update({
        where: { id: bookingId },
        data: {
          paymentStatus: 'PAID',
          momoTransId:   transId,
          paidAmount:    parseInt(amount),
          paidAt:        new Date(),
        }
      }),
      sendConfirmationEmail(bookingId), // gửi email xác nhận
      sendSMS(bookingId),               // gửi SMS
    ]);

    console.log(`✅ Payment success: ${orderId}, transId: ${transId}`);
  } else {
    // ❌ Thanh toán thất bại
    await db.booking.update({
      where: { id: bookingId },
      data: { paymentStatus: 'FAILED', failReason: message }
    });
    console.warn(`❌ Payment failed: ${orderId}, code: ${resultCode}`);
  }

  // QUAN TRỌNG: Phải trả 200 OK trong 30s, MoMo retry 5 lần nếu không nhận được
  res.json({ message: 'OK' });
});

Kiểm tra trạng thái giao dịch node.js 
// POST /v2/gateway/api/query
async function queryTransaction(orderId) {
  const requestId = uuidv4();
  const rawSignature = `accessKey=${MOMO_CONFIG.accessKey}&orderId=${orderId}&partnerCode=${MOMO_CONFIG.partnerCode}&requestId=${requestId}`;
  const signature = crypto.createHmac('sha256', MOMO_CONFIG.secretKey).update(rawSignature).digest('hex');

  const { data } = await axios.post(
    `https://${MOMO_CONFIG.apiUrl}/v2/gateway/api/query`,
    { partnerCode: MOMO_CONFIG.partnerCode, requestId, orderId, signature, lang: 'vi' }
  );
  return data; // resultCode === 0: thành công, khác: thất bại
}
Hoàn tiền khi khách huỷ tour node.js 
// POST /v2/gateway/api/refund
async function refundPayment(orderId, transId, amount, reason) {
  const requestId    = uuidv4();
  const refundOrderId = `REFUND-${orderId}-${Date.now()}`;
  const rawSignature  = `accessKey=${MOMO_CONFIG.accessKey}&amount=${amount}&description=${reason}&orderId=${refundOrderId}&partnerCode=${MOMO_CONFIG.partnerCode}&requestId=${requestId}&transId=${transId}`;
  const signature    = crypto.createHmac('sha256', MOMO_CONFIG.secretKey).update(rawSignature).digest('hex');

  const { data } = await axios.post(
    `https://${MOMO_CONFIG.apiUrl}/v2/gateway/api/refund`,
    { partnerCode: MOMO_CONFIG.partnerCode, requestId, orderId: refundOrderId, transId: String(transId), amount: String(amount), description: reason, signature, lang: 'vi' }
  );
  if (data.resultCode !== 0) throw new Error(`Refund failed: ${data.message}`);
  return data;
}

// Dùng trong route huỷ tour:
router.post('/cancel/:bookingId', async (req, res) => {
  const booking = await db.booking.findUnique({ where: { id: req.params.bookingId } });
  const refund = await refundPayment(
    booking.momoOrderId, booking.momoTransId,
    booking.paidAmount, 'Khách huỷ tour trước 7 ngày'
  );
  res.json({ success: true, refundTransId: refund.transId });
});
