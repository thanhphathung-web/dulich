VNPay dùng HMAC-SHA512 (khác MoMo dùng SHA256)
Tất cả params phải được sort theo alphabet, nối bằng &, hash bằng HMAC-SHA512 với hash secret. Không được có space hay ký tự đặc biệt chưa encode.
utils/vnpay-signature.ts
import crypto from 'crypto';
import qs from 'qs';

/**
 * Tạo VNPay signature — HMAC-SHA512
 * Params phải được sort theo key alphabet A→Z
 * Loại bỏ vnp_SecureHash và vnp_SecureHashType trước khi hash
 */
export function createVNPaySignature(params: Record<string, string>, secretKey: string): string {
  const sortedKeys = Object.keys(params)
    .filter(k => k !== 'vnp_SecureHash' && k !== 'vnp_SecureHashType')
    .sort();

  const rawData = sortedKeys
    .map(k => `${k}=${params[k]}`)
    .join('&');

  return crypto
    .createHmac('sha512', secretKey)
    .update(Buffer.from(rawData, 'utf-8'))
    .digest('hex');
}

/**
 * Verify signature từ VNPay returnUrl params
 * Dùng để xác thực sau khi user redirect về
 */
export function verifyVNPaySignature(
  params: Record<string, string>,
  secretKey: string
): boolean {
  const receivedHash = params['vnp_SecureHash'];
  if (!receivedHash) return false;

  const paramsWithoutHash = { ...params };
  delete paramsWithoutHash['vnp_SecureHash'];
  delete paramsWithoutHash['vnp_SecureHashType'];

  const expectedHash = createVNPaySignature(paramsWithoutHash, secretKey);
  return receivedHash === expectedHash;
}

/** Format date → YYYYMMDDHHmmss theo VNPay spec */
export function vnpayDateFormat(date: Date): string {
  return date.toISOString()
    .replace(/[-T:Z.]/g, '')
    .slice(0, 14);
}
routes/payment/vnpay.ts — Tạo link thanh toán VNPay
import { Router } from 'express';
import qs from 'qs';
import axios from 'axios';
import { createVNPaySignature, verifyVNPaySignature, vnpayDateFormat } from '../../utils/vnpay-signature';
import { authenticate } from '../../middleware/auth';

const router = Router();
const CFG = {
  tmnCode:   process.env.VNPAY_TMN_CODE!,
  hashSecret: process.env.VNPAY_HASH_SECRET!,
  url:       process.env.VNPAY_URL!,
  returnUrl: process.env.VNPAY_RETURN_URL!,
  apiUrl:    process.env.VNPAY_API_URL!,
};

// POST /api/payment/vnpay/create
router.post('/create', authenticate, async (req, res, next) => {
  try {
    const { bookingId, amount, tourName, bankCode } = req.body;

    if (!bookingId || !amount || amount < 10000)
      return res.status(400).json({ error: 'Dữ liệu không hợp lệ' });

    const txnRef = `BT${Date.now()}`; // mã giao dịch unique
    const now    = vnpayDateFormat(new Date());
    const expire = vnpayDateFormat(new Date(Date.now() + 15 * 60 * 1000)); // 15 phút
    const ip     = (req.headers['x-forwarded-for'] || req.socket.remoteAddress || '127.0.0.1') as string;

    const params: Record<string, string> = {
      vnp_Version:    '2.1.0',
      vnp_Command:    'pay',
      vnp_TmnCode:    CFG.tmnCode,
      vnp_Amount:     String(amount * 100), // VNPay nhân 100 (đơn vị xu)
      vnp_CurrCode:   'VND',
      vnp_TxnRef:     txnRef,
      vnp_OrderInfo:  encodeURIComponent(`Dat coc tour ${tourName} - ${bookingId}`),
      vnp_OrderType:  'other', // du lịch dùng 'other'
      vnp_Locale:     'vn',
      vnp_ReturnUrl:  CFG.returnUrl,
      vnp_IpAddr:     ip.split(',')[0].trim(),
      vnp_CreateDate: now,
      vnp_ExpireDate: expire,
      ...(bankCode && { vnp_BankCode: bankCode }), // pre-chọn ngân hàng
    };

    params.vnp_SecureHash = createVNPaySignature(params, CFG.hashSecret);

    // Lưu txnRef → bookingId vào DB để lookup sau khi redirect về
    await db.payment.create({
      data: {
        booking_id:          bookingId,
        provider:            'VNPAY',
        payment_type:        'DEPOSIT',
        provider_request_id: txnRef,
        amount:              amount,
        status:              'PENDING',
        expires_at:          new Date(Date.now() + 15 * 60 * 1000),
      },
    });

    const payUrl = `${CFG.url}?${qs.stringify(params, { encode: false })}`;
    res.json({ payUrl, txnRef });

  } catch (e) { next(e); }
});

export default router;

routes/payment/vnpay.ts — Return URL handler
// GET /api/payment/vnpay/return  ← vnp_ReturnUrl
router.get('/return', async (req, res, next) => {
  try {
    const params = req.query as Record<string, string>;

    // 1. Verify signature — QUAN TRỌNG NHẤT
    const isValid = verifyVNPaySignature(params, CFG.hashSecret);
    if (!isValid) {
      return res.redirect(`${process.env.APP_URL}/payment/failed?reason=invalid_signature`);
    }

    const { vnp_TxnRef, vnp_ResponseCode, vnp_Amount, vnp_TransactionNo, vnp_BankCode } = params;

    // 2. Idempotency — tránh xử lý 2 lần nếu user refresh
    const payment = await db.payment.findUnique({
      where: { provider_request_id: vnp_TxnRef },
    });
    if (!payment) return res.redirect(`${process.env.APP_URL}/payment/failed?reason=not_found`);
    if (payment.status === 'SUCCESS') return res.redirect(`${process.env.APP_URL}/booking/${payment.booking_id}/success`);

    if (vnp_ResponseCode === '00') {
      // 3. Double-check với VNPay query API
      const queryResult = await queryVNPayTransaction(vnp_TxnRef);
      if (queryResult.vnp_TransactionStatus !== '00') {
        return res.redirect(`${process.env.APP_URL}/payment/failed?ref=${vnp_TxnRef}`);
      }

      // 4. Cập nhật DB trong transaction
      await db.$transaction(async tx => {
        await tx.payment.update({
          where: { provider_request_id: vnp_TxnRef },
          data: {
            status:              'SUCCESS',
            provider_trans_id:   vnp_TransactionNo,
            webhook_payload:     params as any,
            paid_at:             new Date(),
          },
        });
        await tx.booking.update({
          where: { id: payment.booking_id },
          data:  { payment_status: 'DEPOSIT_PAID', status: 'CONFIRMED', confirmed_at: new Date() },
        });
      });

      // 5. Gửi email xác nhận
      const booking = await db.booking.findUnique({
        where: { id: payment.booking_id },
        include: { user: true, tour: true },
      });
      sendPaymentSuccess(booking!, { amount: +vnp_Amount / 100, transId: vnp_TransactionNo })
        .catch(console.error);

      return res.redirect(`${process.env.APP_URL}/booking/${payment.booking_id}/success`);
    } else {
      await db.payment.update({
        where: { provider_request_id: vnp_TxnRef },
        data: { status: 'FAILED', failure_reason: `VNPay code: ${vnp_ResponseCode}` },
      });
      return res.redirect(`${process.env.APP_URL}/payment/failed?code=${vnp_ResponseCode}`);
    }
  } catch (e) { next(e); }
});

// Query VNPay API để verify giao dịch
async function queryVNPayTransaction(txnRef: string) {
  const now    = vnpayDateFormat(new Date());
  const params = {
    vnp_Version:   '2.1.0',
    vnp_Command:   'querydr',
    vnp_TmnCode:   CFG.tmnCode,
    vnp_TxnRef:    txnRef,
    vnp_OrderInfo: `Query transaction ${txnRef}`,
    vnp_TransDate: now,
    vnp_CreateDate: now,
    vnp_IpAddr:   '127.0.0.1',
  };
  params['vnp_SecureHash'] = createVNPaySignature(params, CFG.hashSecret);
  const { data } = await axios.post(CFG.apiUrl, params);
  return data;
}

// POST /api/payment/vnpay/refund — admin only
router.post('/refund', authenticate, requireRole('ADMIN'), async (req, res, next) => {
  try {
    const { txnRef, amount, transDate, reason } = req.body;
    const now = vnpayDateFormat(new Date());
    const refundRef = `RF${Date.now()}`;

    const params = {
      vnp_Version:       '2.1.0',
      vnp_Command:       'refund',
      vnp_TmnCode:       CFG.tmnCode,
      vnp_TransactionType: '02', // 02 = hoàn toàn bộ, 03 = hoàn một phần
      vnp_TxnRef:        refundRef,
      vnp_Amount:        String(amount * 100),
      vnp_OrderInfo:     encodeURIComponent(reason),
      vnp_TransDate:     transDate,  // ngày giao dịch gốc
      vnp_CreateBy:      'admin',
      vnp_CreateDate:    now,
      vnp_IpAddr:        '127.0.0.1',
    };
    params['vnp_SecureHash'] = createVNPaySignature(params, CFG.hashSecret);

    const { data } = await axios.post(CFG.apiUrl, params);
    if (data.vnp_ResponseCode !== '00')
      throw new Error(`VNPay refund failed: ${data.vnp_ResponseCode} — ${data.vnp_Message}`);

    res.json({ success: true, refundRef, data });
  } catch (e) { next(e); }
});
